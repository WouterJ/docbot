# The DocBot

Usage:

    # linting a directory
    $ ./docbot.php lint some_directory/

    # linting a file
    $ ./docbot.php lint path/to/file.rst

    # specifying a reviewer type (available types: rst, doc, symfony)
    $ ./docbot.php lint --types rst ...

    # or multiple types
    $ ./docbot.php lint --types rst,doc ...
